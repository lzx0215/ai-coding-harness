# Reviewer Provenance v0.2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the v0.2 reviewer provenance contract while preserving v0.1 and v0.1.1 run compatibility.

**Architecture:** Use a two-phase implementation order. First, land additive state schema compatibility for `0.1.0|0.2.0` and prove existing run records still validate. Then add optional structured review provenance to the Claude review output schema and adapter, preserving scalar compatibility fields for old consumers.

**Tech Stack:** Python 3.12, `unittest`, `jsonschema` Draft 2020-12, JSON Schema, Markdown, Git.

---

## Scope Check

This plan implements one cohesive v0.2 slice: reviewer provenance, nullable metadata consumer compatibility, and additive state schema version compatibility.

It does not add async review jobs, change Claude permissions, add a UI, migrate historical runs in place, or change authentication and production configuration.

## Commit Strategy

Use two main implementation commits before final review cleanup:

1. `test(schema): allow v0.2 harness state versions`
   - Only state schema compatibility and tests.
   - Validate existing v0.1 and v0.1.1 run records before moving on.
2. `feat(review): add reviewer provenance metadata`
   - Output schema, adapter provenance extraction, fixtures, templates, and docs.
   - Adapter may emit `reviewer_provenance.schema_version = "0.2.0"` only after Commit 1 exists.

A final documentation or run-evidence commit is allowed if implementation evidence needs to be separated from code changes.

    text = _optional_string(raw_version)
    if text is None:

Add:

def complete_usage_total(raw_usage: Any) -> int | None:
    if not isinstance(raw_usage, dict):
        return None
    input_tokens = raw_usage.get("inputTokens")
    output_tokens = raw_usage.get("outputTokens")
    if isinstance(input_tokens, bool) or isinstance(output_tokens, bool):
        return None
    if isinstance(input_tokens, int) and isinstance(output_tokens, int):
        if input_tokens >= 0 and output_tokens >= 0:
            return input_tokens + output_tokens
    return None


def select_primary_model(model_usage: Any) -> str | None:
    if not isinstance(model_usage, dict):
        return None

    names = sorted(
        name.strip()
        for name in model_usage
        if isinstance(name, str) and name.strip()
    )
    if not names:
        return None

    complete: list[tuple[int, str]] = []
    for name in names:
        total = complete_usage_total(model_usage.get(name))
        if total is not None:
            complete.append((total, name))

    if complete:
        largest_total = max(total for total, _name in complete)
        return sorted(name for total, name in complete if total == largest_total)[0]

    return names[0]
```

    ordered_names = model_usage_names(model_usage)
    if primary in ordered_names:
        ordered_names.remove(primary)
        ordered_names.insert(0, primary)

    entries: list[dict[str, Any]] = []
    for name in ordered_names:
        entries.append(
            {
                "name": name,
                "version": None,
                "source": "modelUsage",
                "usage": {
                    "input_tokens": None,
                    "output_tokens": None,
                },
                "raw_usage": model_usage.get(name),
            }
        )
    return entries


def metadata_model_entry(name: str) -> dict[str, Any]:
    return {
        "name": name,
        "version": None,
        "source": "metadata",
        "usage": {
            "input_tokens": None,
            "output_tokens": None,
        },
    }


def build_reviewer_provenance(
    parsed: dict[str, Any],
    content: dict[str, Any],
    *,
    reviewer_cli_version: str | None,
) -> dict[str, Any]:
    model_usage = parsed.get("modelUsage")
    explicit_model = _metadata_value(parsed, content, "reviewer_model", "model")
    unknowns: set[str] = {"token_usage"}
    models: list[dict[str, Any]] = []

    if explicit_model is not None:
        primary = explicit_model
        models.append(metadata_model_entry(explicit_model))
        for entry in model_usage_model_entries(
            model_usage,
            primary=select_primary_model(model_usage),
        ):
            if entry["name"] != explicit_model:
                models.append(entry)
    elif isinstance(model_usage, dict):
        primary = select_primary_model(model_usage)
        models.extend(model_usage_model_entries(model_usage, primary=primary))
    else:
        primary = None

    if models:
        unknowns.add("model_version")
    else:
        unknowns.update({"model_name", "primary_model"})

    if reviewer_cli_version is None:
        unknowns.add("cli_version")

    return {
        "schema_version": PROVENANCE_SCHEMA_VERSION,
        "reviewer": REVIEWER,
        "cli": {
            "name": "Claude Code",
            "raw_version": _optional_string(reviewer_cli_version),
            "version": parse_cli_version(reviewer_cli_version),
        },
        "models": models,
        "primary_model": primary,
        "unknowns": sorted(unknowns),
    }
```
